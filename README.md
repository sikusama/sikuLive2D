<script>
    function calculateTotal() {
        let total = 0;
        const baseType = document.querySelector('input[name="rigging-type"]:checked');
        if (baseType) total += parseInt(baseType.value);

        const extras = document.querySelectorAll('.extra-option:checked');
        extras.forEach(extra => { total += parseInt(extra.value); });

        const collab = document.querySelector('input[name="collab"]:checked');
        if (collab) total += parseInt(collab.value);

        if (total < 0) total = 0;
        let beforePremium = total;

        const pfFactor = parseFloat(document.querySelector('input[name="portfolio"]:checked').value);
        total = total * pfFactor;

        const isCommercial = document.getElementById('commercial-use').checked;
        if (isCommercial) { total = total * 3; }

        const finalPrice = Math.floor(total);
        document.getElementById('base-sum-display').innerText = "할인 적용 기본 합계: " + beforePremium.toLocaleString() + "원";
        document.getElementById('total-price-display').innerText = "최종 가격: " + finalPrice.toLocaleString() + "원";
    }

    function copyForm() {
        const finalPrice = document.getElementById('total-price-display').innerText;
        const desc = document.getElementById('desc-input').value;
        const notes = document.getElementById('extra-notes').value;
        
        let content = "[리깅 견적 신청서]\n";
        content += "- " + finalPrice + "\n";
        content += "- 설명: " + (desc ? desc : "없음") + "\n";
        content += "- 메모: " + (notes ? notes : "없음");

        const textarea = document.createElement("textarea");
        textarea.value = content;
        document.body.appendChild(textarea);
        textarea.select();
        document.execCommand("copy");
        document.body.removeChild(textarea);
        
        alert("견적 내용이 클립보드에 복사되었습니다. 채팅창에 붙여넣어주세요!");
    }
    
    calculateTotal();
</script>
